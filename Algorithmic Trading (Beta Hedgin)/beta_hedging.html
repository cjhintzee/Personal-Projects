# Calculate portfolio returns with volume-based weights
def calculate_weighted_returns(price_data, avg_volume):
    weights = avg_volume.div(avg_volume.sum(axis=1), axis=0)  # Normalize to get weights
    weighted_returns = (price_data.pct_change() * weights.shift(1)).sum(axis=1)
    return weighted_returns

def hedged_portfolio_performance_rolling_beta(
    portfolio_size,
    rebalancing_days,
    lookback_days,
    benchmark_returns,
    long_returns,
    transaction_cost
):
    if len(long_returns) < lookback_days + rebalancing_days:
        print("Error: Not enough data.")
        return [], [], []

    warmup_benchmark_returns = benchmark_returns[:lookback_days]
    warmup_long_returns = long_returns[:lookback_days]

    X = sm.add_constant(warmup_benchmark_returns)
    model = sm.OLS(warmup_long_returns, X).fit()
    initial_beta = model.params[1]

    portfolio_value = portfolio_size
    portfolio_values = [portfolio_value]
    betas = [initial_beta]
    # --- ADDITION --- Track dates for plotting
    dates = [benchmark_returns.index[lookback_days - 1]]

    long_ratio = 1 / (1 + initial_beta)
    short_ratio = initial_beta / (1 + initial_beta)
    long_allocation = portfolio_value * long_ratio
    short_allocation = portfolio_value * short_ratio

    for i in range(lookback_days, len(long_returns), rebalancing_days):
        start_trade_period = i
        end_trade_period = min(i + rebalancing_days, len(long_returns))

        long_period_returns = long_returns[start_trade_period:end_trade_period]
        short_period_returns = -benchmark_returns[start_trade_period:end_trade_period]

        if len(long_period_returns) > 0:
            long_cum_return = np.prod(1 + long_period_returns) - 1
            short_cum_return = np.prod(1 + short_period_returns) - 1
            
            long_allocation *= (1 + long_cum_return)
            short_allocation *= (1 + short_cum_return)

            portfolio_value = long_allocation + short_allocation
            portfolio_values.append(portfolio_value)
            # --- ADDITION --- Append the date at the end of the trading period
            dates.append(benchmark_returns.index[end_trade_period - 1])

        start_beta_calc = end_trade_period - lookback_days
        end_beta_calc = end_trade_period
        
        if start_beta_calc < 0: continue
            
        lookback_benchmark = benchmark_returns[start_beta_calc:end_beta_calc]
        lookback_long = long_returns[start_beta_calc:end_beta_calc]

        X = sm.add_constant(lookback_benchmark)
        model = sm.OLS(lookback_long, X).fit()
        beta = model.params[1]
        betas.append(beta)

        long_ratio = 1 / (1 + beta)
        short_ratio = beta / (1 + beta)

        long_allocation = portfolio_value * long_ratio * (1 - transaction_cost)
        short_allocation = portfolio_value * short_ratio * (1 - transaction_cost)

    return portfolio_values, betas, dates
